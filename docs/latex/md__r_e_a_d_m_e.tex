A command-\/line personal finance tracker written in C++17. ~\newline
 It lets you set a monthly budget, record categorised expenses, and view spending summaries — all stored in a human-\/readable JSON file that persists between sessions.

\DoxyHorRuler{0}
\hypertarget{md__r_e_a_d_m_e_autotoc_md2}{}\doxysection{\texorpdfstring{Features}{Features}}\label{md__r_e_a_d_m_e_autotoc_md2}

\begin{DoxyItemize}
\item Set and update a monthly budget
\item Add expenses with an amount, category, and date
\item View a full financial summary (budget / spent / remaining)
\item View spending grouped by category
\item View statistics for any date range (YYYY-\/\+MM-\/\+DD)
\item Data stored in a plain-\/text JSON file (path configurable at runtime)
\item Full exception-\/based error handling — no silent failures
\end{DoxyItemize}
