12struct Approx {

14 explicit Approx(double v, double e = 1e-9) : val(v), eps(e) {}
15};
16inline bool operator==(double a, const Approx &b) { return std::abs(a-b.val) <= b.eps; }
17inline bool operator==(const Approx &a, double b) { return b == a; }

20namespace doctest_impl {
21struct TestCase { const char *name; std::function<void()> fn; };
22inline std::vector<TestCase> &registry() { static std::vector<TestCase> r; return r; }
23struct Reg { Reg(const char *n, std::function<void()> f) { registry().push_back({n,f}); } };
24inline int &fails() { static int n = 0; return n; }
25inline const char *&curTest() { static const char *s = ""; return s; }
26inline void check(bool ok, const char *expr, const char *file, int line) {
27 if (ok) return;
28 ++fails();
29 std::cerr << file << ":" << line << ": FAILED [" << curTest() << "]\n"
30 << " CHECK(" << expr << ")\n";
31}

34// Each TEST_CASE gets a unique struct type whose constructor registers the test.
35// This avoids needing __COUNTER__ to appear twice with the same value.
36#define TEST_CASE(name) \
37 static void DOCTEST_FN(__LINE__)(); \
38 namespace { struct DOCTEST_ST(__LINE__) { \
39 DOCTEST_ST(__LINE__)() { \
40 ::doctest_impl::registry().push_back({name, DOCTEST_FN(__LINE__)}); \
41 } \
42 } DOCTEST_OBJ(__LINE__); } \
43 static void DOCTEST_FN(__LINE__)()

45#define DOCTEST_FN(l) DOCTEST_CAT(doctest_fn_, l)
46#define DOCTEST_ST(l) DOCTEST_CAT(doctest_st_, l)
47#define DOCTEST_OBJ(l) DOCTEST_CAT(doctest_obj_, l)
48#define DOCTEST_CAT(a,b) DOCTEST_CAT2(a,b)

54#define CHECK_THROWS_AS(expr, etype) \
